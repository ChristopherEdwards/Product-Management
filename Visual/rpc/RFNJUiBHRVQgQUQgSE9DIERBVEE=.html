RETURN VALUE TYPEGLOBAL ARRAY

DESCRIPTION
GET DATA FOR AN AD HOC REPORT
INPUT PARAMETER
  1. INPUT PARAMETER:
    INARRAY
    PARAMETER TYPE:
    LIST
    REQUIRED:
    YES
    SEQUENCE NUMBER:
    1
    DESCRIPTION:
    Input Parameter(s) - ARRAY AS FOLLOWS:
 "F1_IEN^F2_IEN^...^Fn_IEN
 FILTER/SELECTION FIELDS (OPTIONAL)
 "F1"^F1_IEN^F1_VALUE
 "F2"^F2_IEN^F1_VALUE
 "F3"^F3_IEN^F1_VALUE
 DATE RANGE FIELDS (OPTIONAL)
 "R1"^R1_IEN^F1_VALUE1^F1_VALUE2
 "R2"^R2_IEN^R2_VALUE2^R2_VALUE2^ANDOR (A=AND O=OR DATE RANGES)

 "S1"^S1_IEN^["A"|"D"] ("A - ASCENDING IS THE DEFAULT, "D" - DESCENDING)
 "S2"^S2_IEN^["A"|"D"] ("A - ASCENDING IS THE DEFAULT, "D" - DESCENDING)
RETURN PARAMETER DESCRIPTION
Return Value
 ARRAY:- VARIABLE NUMBER OF RECORDS AS FOLLOWS:
